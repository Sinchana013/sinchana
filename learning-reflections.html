Business Case 1: BRTS ticket scanning

Description: Verifies ticket using QR codes, RFID cards, or physical passes for efficient and secure travel.

  • As the tickets are scanned in, they are put into a heap and then while scanning at the entrance, we use heap's search function to check if the person has a valid ticket.

  • We could have also used other data structures, but we have chosen a heap for best results.

  • Search is done efficiently in heaps compared to other data structures.

  • Whenever the person scans out, their ticket node should be deleted.

  • Heap provides a very efficient deletion function, making it suitable for this case.

Code for Business Case 1

#include 
                    #include 
                    #include 
                    
                    using namespace std;
                    
                    // Define the structure for the heap
                    class TicketHeap {
                    private:
                        vector heap;  // Vector to store the heap (ticket IDs)
                    
                        // Helper function to maintain the heap property (min-heap)
                        void heapify_up(int index) {
                            while (index > 0 && heap[index] < heap[(index - 1) / 2]) {
                                swap(heap[index], heap[(index - 1) / 2]);
                                index = (index - 1) / 2;
                            }
                        }
                    
                        void heapify_down(int index) {
                            int leftChild = 2 * index + 1;
                            int rightChild = 2 * index + 2;
                            int smallest = index;
                    
                            if (leftChild < heap.size() && heap[leftChild] < heap[smallest]) {
                                smallest = leftChild;
                            }
                    
                            if (rightChild < heap.size() && heap[rightChild] < heap[smallest]) {
                                smallest = rightChild;
                            }
                    
                            if (smallest != index) {
                                swap(heap[index], heap[smallest]);
                                heapify_down(smallest);
                            }
                        }
                    
                    public:
                        // Function to scan in and add ticket to the heap
                        void scanIn(int ticketId) {
                            heap.push_back(ticketId);
                            heapify_up(heap.size() - 1);
                            cout << "Ticket " << ticketId << " scanned in successfully." << endl;
                        }
                    
                        // Function to scan out and remove ticket from the heap
                        void scanOut(int ticketId) {
                            auto it = find(heap.begin(), heap.end(), ticketId);
                            
                            if (it == heap.end()) {
                                cout << "Access Denied" << endl;
                                return;
                            }
                    
                            // Remove the ticket from heap
                            int index = it - heap.begin();
                            swap(heap[index], heap[heap.size() - 1]);
                            heap.pop_back();
                            heapify_down(index);
                    
                            cout << "Ticket " << ticketId << " scanned out successfully." << endl;
                        }
                    
                        // Display the heap for debugging or testing
                        void displayHeap() {
                            cout << "Current Heap: ";
                            for (int i = 0; i < heap.size(); ++i) {
                                cout << heap[i] << " ";
                            }
                            cout << endl;
                        }
                    };
                    
                    int main() {
                        TicketHeap ticketSystem;
                    
                        // Simulating ticket scanning
                        ticketSystem.scanIn(101); // Scanning in ticket 101
                        ticketSystem.scanIn(102); // Scanning in ticket 102
                        ticketSystem.scanIn(103); // Scanning in ticket 103
                    
                        ticketSystem.displayHeap();
                    
                        ticketSystem.scanOut(102); // Scanning out ticket 102
                        ticketSystem.displayHeap();
                    
                        ticketSystem.scanOut(105); // Attempting to scan out non-existent ticket 105
                        ticketSystem.displayHeap();
                    
                        return 0;
                    }

Business Case 2: Booked Ticket Search

Description: A search system to allow users to retrieve previously booked ticket details.

  • Here the task is to retrieve the data based on ticket ID.

  • Why KMP? It's not the case that we can only implement KMP for this solution. We can use many other algorithms we studied too.

  • But the vital property we need is its efficiency and ability to match strings for larger string lengths.

  • KMP ensures efficient search even with large datasets by avoiding unnecessary re-comparisons.

  • KMP handles partial searches efficiently by utilizing its LPS array for faster matching.

  • So clearly, KMP algorithm has an upper hand in solving the task.

                    using namespace std;
                    
                    // Function to compute the LPS (Longest Prefix Suffix) array for KMP algorithm
                    vector computeLPSArray(const string &pattern) {
                        int m = pattern.length();
                        vector lps(m, 0);
                        int length = 0; // Length of the previous longest prefix suffix
                        int i = 1;
                    
                        while (i < m) {
                            if (pattern[i] == pattern[length]) {
                                length++;
                                lps[i] = length;
                                i++;
                            } else {
                                if (length != 0) {
                                    length = lps[length - 1];
                                } else {
                                    lps[i] = 0;
                                    i++;
                                }
                            }
                        }
                        return lps;
                    }
                    
                    // KMP String Matching Algorithm
                    bool KMPSearch(const string &text, const string &pattern) {
                        int n = text.length();
                        int m = pattern.length();
                        vector lps = computeLPSArray(pattern);
                    
                        int i = 0; // Index for text
                        int j = 0; // Index for pattern
                        while (i < n) {
                            if (pattern[j] == text[i]) {
                                i++;
                                j++;
                            }
                            if (j == m) {
                                return true; // Pattern found in text
                            } else if (i < n && pattern[j] != text[i]) {
                                if (j != 0) {
                                    j = lps[j - 1];
                                } else {
                                    i++;
                                }
                            }
                        }
                        return false;
                    }
                    
                    // Structure to store ticket details
                    struct Ticket {
                        string ticketID;
                        string userName;
                        string travelDate;
                        string status; // e.g., confirmed, canceled
                    };
                    
                    // Function to search tickets
                    void searchTickets(const vector &tickets, const string &query) {
                        cout << "Search Results for \"" << query << "\":\n";
                        bool found = false;
                    
                        for (const auto &ticket : tickets) {
                            if (KMPSearch(ticket.ticketID, query) || KMPSearch(ticket.userName, query) || KMPSearch(ticket.travelDate, query)) {
                                found = true;
                                cout << "Ticket ID: " << ticket.ticketID << "\n"
                                     << "User Name: " << ticket.userName << "\n"
                                     << "Travel Date: " << ticket.travelDate << "\n"
                                     << "Status: " << ticket.status << "\n\n";
                            }
                        }
                    
                        if (!found) {
                            cout << "No tickets match the search query.\n";
                        }
                    }
                    
                    int main() {
                        // Sample ticket data
                       // Sample ticket data
                        vector tickets = {
                            {"T12345", "Nisha", "2024-12-01", "Confirmed"},
                            {"T67890", "Deepika", "2024-12-15", "Canceled"},
                            {"T54321", "Namrutha", "2024-12-20", "Confirmed"},
                            {"T98765", "Sinchana", "2024-12-25", "Confirmed"}
                        };
                    
                        // Hardcoded search query
                        string query = "Alice";
                    
                        searchTickets(tickets, query);
                    
                        return 0;
                    }
