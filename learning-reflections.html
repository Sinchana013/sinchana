Business Case 1: BRTS ticket scanning

Description: Verifies ticket using QR codes, RFID cards, or physical passes for efficient and secure travel.

  • As the tickets are scanned in, they are put into a heap and then while scanning at the entrance, we use heap's search function to check if the person has a valid ticket.

  • We could have also used other data structures, but we have chosen a heap for best results.

  • Search is done efficiently in heaps compared to other data structures.

  • Whenever the person scans out, their ticket node should be deleted.

  • Heap provides a very efficient deletion function, making it suitable for this case.

  • Note : Both code implementation and output are provided in the code edittor below, please do scroll to view the outputt.

// Node for the BST
struct BSTNode {
    Ticket ticket;
    BSTNode* left;
    BSTNode* right;

    BSTNode(Ticket t) : ticket(t), left(NULL), right(NULL) {}
};

// Queue for Tickets and Passes
queue ticketQueue;

// BST Root
BSTNode* root = NULL;

// Static Date and Time (Example)
string staticDateTime = "2025-01-02";  // Example static date

// Add Ticket to BST
BSTNode* insertIntoBST(BSTNode* root, Ticket ticket) {
    if (!root) return new BSTNode(ticket);
    if (ticket.id < root->ticket.id)
        root->left = insertIntoBST(root->left, ticket);
    else if (ticket.id > root->ticket.id)
        root->right = insertIntoBST(root->right, ticket);
    return root;
}

// Search in BST
BSTNode* searchInBST(BSTNode* root, int id) {
    if (!root || root->ticket.id == id) return root;
    if (id < root->ticket.id)
        return searchInBST(root->left, id);
    return searchInBST(root->right, id);
}

// Delete from BST
BSTNode* deleteFromBST(BSTNode* root, int id) {
    if (!root) return root;
    if (id < root->ticket.id)
        root->left = deleteFromBST(root->left, id);
    else if (id > root->ticket.id)
        root->right = deleteFromBST(root->right, id);
    else {
        if (!root->left) {
            BSTNode* temp = root->right;
            delete root;
            return temp;
        } else if (!root->right) {
            BSTNode* temp = root->left;
            delete root;
            return temp;
        }

        BSTNode* temp = root->right;
        while (temp->left) temp = temp->left;
        root->ticket = temp->ticket;
        root->right = deleteFromBST(root->right, temp->ticket.id);
    }
    return root;
}

void scanIn() {

        int id;
        cout << "Enter Ticket ID: ";

        // Search for the ticket in the BST
        BSTNode* existingNode = searchInBST(root, id);

        if (existingNode) {
            // If the ticket is found and it's already scanned-in
            if (existingNode->ticket.status == "scanned-in") {
                cout << "Ticket already scanned in. No entry allowed!" << endl;
            } else {
                // Ticket is found and was scanned-out, allow scanning in again
                existingNode->ticket.status = "scanned-in";  // Change status to "scanned-in"
                cout << "Ticket scanned in successfully. Entry allowed!" << endl;
            }
        } else {
            // Ticket not found in BST, allowing the ticket to be added and scanned in
            Ticket t;
            t.id = id;
            t.dateTime = staticDateTime;  // Static date
            t.source = "source";
            t.destination = "destination";
            t.status = "scanned-in";
            root = insertIntoBST(root, t);
            cout << "Ticket scanned in successfully. Entry allowed!" << endl;
        }
    } else if (type == "pass") {
        int passId;
        cout << "Enter Pass ID: ";
        cin >> passId;

        bool found = false;
        queue tempQueue = passQueue;
        while (!tempQueue.empty()) {
            if (tempQueue.front().passId == passId) {
                found = true;
                break;
            }
            tempQueue.pop();
        }

        if (found) {
            cout << "Pass is valid. Entry allowed!" << endl;
        } else {
            cout << "Invalid pass. No entry allowed!" << endl;
        }
    } else {
        cout << "Invalid input. Try again." << endl;
    }
}

Business Case 2: Booked Ticket Search

Description: A search system to allow users to retrieve previously booked ticket details.

  • Here the task is to retrieve the data based on ticket ID.

  • Why KMP? It's not the case that we can only implement KMP for this solution. We can use many other algorithms we studied too.

  • But the vital property we need is its efficiency and ability to match strings for larger string lengths.

  • KMP ensures efficient search even with large datasets by avoiding unnecessary re-comparisons.

  • KMP handles partial searches efficiently by utilizing its LPS array for faster matching.

  • So clearly, KMP algorithm has an upper hand in solving the task.

                    // Function to compute the LPS (Longest Prefix Suffix) array for KMP algorithm
                    vector computeLPSArray(const string &pattern) {
                        int m = pattern.length();
                        vector lps(m, 0);
                        int length = 0; // Length of the previous longest prefix suffix
                        int i = 1;
                    
                        while (i < m) {
                            if (pattern[i] == pattern[length]) {
                                length++;
                                lps[i] = length;
                                i++;
                            } else {
                                if (length != 0) {
                                    length = lps[length - 1];
                                } else {
                                    lps[i] = 0;
                                    i++;
                                }
                            }
                        }
                        return lps;
                    }
                    
                    // KMP String Matching Algorithm
                    bool KMPSearch(const string &text, const string &pattern) {
                        int n = text.length();
                        int m = pattern.length();
                        vector lps = computeLPSArray(pattern);
                    
                        int i = 0; // Index for text
                        int j = 0; // Index for pattern
                        while (i < n) {
                            if (pattern[j] == text[i]) {
                                i++;
                                j++;
                            }
                            if (j == m) {
                                return true; // Pattern found in text
                            } else if (i < n && pattern[j] != text[i]) {
                                if (j != 0) {
                                    j = lps[j - 1];
                                } else {
                                    i++;
                                }
                            }
                        }
                        return false;
                    }
                    
                    // Structure to store ticket details
                    struct Ticket {
                        string ticketID;
                        string userName;
                        string travelDate;
                        string status; // e.g., confirmed, canceled
                    };
                    
                    // Function to search tickets
                    void searchTickets(const vector &tickets, const string &query) {
                        cout << "Search Results for \"" << query << "\":\n";
                        bool found = false;
                    
                        for (const auto &ticket : tickets) {
                            if (KMPSearch(ticket.ticketID, query) || KMPSearch(ticket.userName, query) || KMPSearch(ticket.travelDate, query)) {
                                found = true;
                                cout << "Ticket ID: " << ticket.ticketID << "\n"
                                     << "User Name: " << ticket.userName << "\n"
                                     << "Travel Date: " << ticket.travelDate << "\n"
                                     << "Status: " << ticket.status << "\n\n";
                            }
                        }
                    
                        if (!found) {
                            cout << "No tickets match the search query.\n";
                        }
                    }
