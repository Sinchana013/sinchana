                using namespace std;
                
                // Passenger class to store details
                class Passenger {
                private:
                    string name;
                    string source;
                    string destination;
                    double ticketPrice;
                
                public:
                    // Constructor
                    Passenger(string pname, string psource, string pdestination, double pprice)
                        : name(pname), source(psource), destination(pdestination), ticketPrice(pprice) {}
                
                    // Function to display passenger details (now marked as const)
                    void displayDetails() const {
                        cout << left << setw(15) << name
                             << setw(15) << source
                             << setw(15) << destination
                             << fixed << setprecision(2) << "Rs. " << ticketPrice << endl;
                    }
                };
                
                // Main function
                int main() {
                    vector passengers; // Vector to store passengers
                    char choice;
                
                    do {
                        // Input details from the user
                        string name, source, destination;
                        double ticketPrice;
                
                        cout << "Enter Passenger Name: ";
                        cin.ignore(numeric_limits::max(), '\n'); // Clear input buffer before getline
                        getline(cin, name);
                
                        cout << "Enter Source: ";
                        getline(cin, source);
                
                        cout << "Enter Destination: ";
                        getline(cin, destination);
                
                        cout << "Enter Ticket Price: Rs. ";
                        cin >> ticketPrice;
                
                        // Clear the buffer before the next iteration or getline
                        cin.ignore(numeric_limits::max(), '\n');
                
                        // Create a passenger object and add it to the list
                        Passenger passenger(name, source, destination, ticketPrice);
                        passengers.push_back(passenger);
                
                        cout << "Do you want to add another passenger? (y/n): ";
                        cin >> choice;
                
                    } while (choice == 'y' || choice == 'Y');
                
                    // Display all passengers
                    cout << "\n--- Passenger List ---" << endl;
                    cout << left << setw(15) << "Name"
                         << setw(15) << "Source"
                         << setw(15) << "Destination"
                         << "Ticket Price" << endl;
                    cout << string(60, '-') << endl;
                
                    for (const auto& passenger : passengers) {
                        passenger.displayDetails(); // Works because displayDetails is now const
                    }
                
                    return 0;
                }
